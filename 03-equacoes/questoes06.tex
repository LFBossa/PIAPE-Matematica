\paragraph*{1.} Resolva as inequações irracionais:
\begin{enumerate}[a)]     
    \item $\sqrt{3x-2} < 2$ % 554a [2/3,2)
    \item $\sqrt{4-3x} \le x$ %555a [1,4/3]
    \item $\sqrt{2x+9} < x-3 $ %555c x > 8
    \item $\sqrt{2x^2 - x - 6} \le x$ % 555f [2,3]
    \item $\sqrt{2x+3} > 5$ % 558a x > 11
    \item $\sqrt{3x-2} > x$ % 559a 1 < x < 2
    \item $\sqrt{6x^2 + x -1} > 2x +1$ % 559d x < -1/2 ou x > 2
\end{enumerate}
