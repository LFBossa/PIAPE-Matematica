\textbf{Exercícios Aula 04}         
\begin{multicols}{2}
\paragraph*{1.} Resolva as inequações de segundo grau abaixo
\begin{enumerate}[a)]    
    \item $x^2 - x - 2 > 0$ 
    \item $x^2 -7x + 12 \le 0$
    \item $x^2 - 5x + 6 < 0$
    \item $-x^2 + 6x  - 9  \ge 0$
    \item $-2x^2 -3x + 2 \le 0$ 
    \item $6x^2 + x - 15 \le 0$
    \item $-2x^2 + 8 \ge 0$
\end{enumerate}
 \vspace*{6cm}
\end{multicols}
